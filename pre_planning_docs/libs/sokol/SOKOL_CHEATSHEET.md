## 1. Minimal App Template

```c
#include "sokol_app.h"
#include "sokol_gfx.h"
#include "sokol_glue.h"
#include "sokol_log.h"
#include "sokol_time.h"
// #include "shaders/my_shader.glsl.h"   // generated by sokol-shdc

static struct {
    sg_pipeline pip;
    sg_bindings bind;
    sg_pass_action pass_action;
    uint64_t last_time;
} state;

static void init(void) {
    sg_setup(&(sg_desc){
        .environment = sglue_environment(),
        .logger.func = slog_func,
    });
    stm_setup();

    state.pass_action = (sg_pass_action){
        .colors[0] = {
            .load_action = SG_LOADACTION_CLEAR,
            .clear_value = { 0.1f, 0.1f, 0.1f, 1.0f },
        },
    };
    // ... create buffers, shaders, pipelines ...
}

static void frame(void) {
    uint64_t dt = stm_laptime(&state.last_time);
    double dt_sec = stm_sec(dt);

    sg_begin_pass(&(sg_pass){
        .action = state.pass_action,
        .swapchain = sglue_swapchain(),
    });
    sg_apply_pipeline(state.pip);
    sg_apply_bindings(&state.bind);
    // sg_apply_uniforms(0, &SG_RANGE(vs_params));
    // sg_draw(0, num_elements, 1);
    sg_end_pass();
    sg_commit();
}

static void cleanup(void) {
    sg_shutdown();
}

static void event(const sapp_event* ev) {
    // handle input
}

sapp_desc sokol_main(int argc, char* argv[]) {
    (void)argc; (void)argv;
    return (sapp_desc){
        .init_cb    = init,
        .frame_cb   = frame,
        .cleanup_cb = cleanup,
        .event_cb   = event,
        .width      = 1280,
        .height     = 720,
        .window_title = "My App",
        .logger.func  = slog_func,
    };
}
```

## 2. sokol_gfx.h — Graphics Core

### 2.1 Resource ID Types

```c
typedef struct sg_buffer   { uint32_t id; } sg_buffer;
typedef struct sg_image    { uint32_t id; } sg_image;
typedef struct sg_sampler  { uint32_t id; } sg_sampler;
typedef struct sg_shader   { uint32_t id; } sg_shader;
typedef struct sg_pipeline { uint32_t id; } sg_pipeline;
typedef struct sg_view     { uint32_t id; } sg_view;
```

All resource IDs wrap a `uint32_t`. An ID of `0` (`SG_INVALID_ID`) means invalid/uninitialized.

### 2.2 Compile-Time Constants

| Constant | Value | Meaning |
|----------|-------|---------|
| `SG_INVALID_ID` | 0 | Invalid resource ID |
| `SG_NUM_INFLIGHT_FRAMES` | 2 | Double-buffered frames |
| `SG_MAX_COLOR_ATTACHMENTS` | 8 | Max MRT targets |
| `SG_MAX_VERTEX_ATTRIBUTES` | 16 | Max vertex attrs per pipeline |
| `SG_MAX_VERTEXBUFFER_BINDSLOTS` | 8 | Max vertex buffer bind slots |
| `SG_MAX_UNIFORMBLOCK_BINDSLOTS` | 8 | Max UB bind slots |
| `SG_MAX_UNIFORMBLOCK_MEMBERS` | 16 | Max members in a UB |
| `SG_MAX_VIEW_BINDSLOTS` | 32 | Max view (texture/storage) bind slots |
| `SG_MAX_SAMPLER_BINDSLOTS` | 12 | Max sampler bind slots |
| `SG_MAX_TEXTURE_SAMPLER_PAIRS` | 32 | Max texture+sampler pairs |
| `SG_MAX_MIPMAPS` | 16 | Max mipmap levels |

### 2.3 Utility Types

```c
typedef struct sg_range { const void* ptr; size_t size; } sg_range;
```

```c
// C:   SG_RANGE(my_data)    → (sg_range){ &my_data, sizeof(my_data) }
```

## 4. sokol_time.h — Time Measurement

### 4.1 All Functions

```c
void     stm_setup(void);                             // init (call once)
uint64_t stm_now(void);                               // current time in ticks
uint64_t stm_diff(uint64_t new_ticks, uint64_t old);  // new - old (always positive)
uint64_t stm_since(uint64_t start);                   // stm_diff(stm_now(), start)
uint64_t stm_laptime(uint64_t* last_time);             // frame delta helper
uint64_t stm_round_to_common_refresh_rate(uint64_t t); // snap to 60/120/144Hz etc.

double stm_sec(uint64_t ticks);                        // ticks → seconds
double stm_ms(uint64_t ticks);                         // ticks → milliseconds
double stm_us(uint64_t ticks);                         // ticks → microseconds
double stm_ns(uint64_t ticks);                         // ticks → nanoseconds
```

### 4.2 Usage Patterns

```c
// --- Basic elapsed time ---
stm_setup();
uint64_t start = stm_now();
// ... work ...
double elapsed_ms = stm_ms(stm_since(start));

// --- Frame delta time ---
static uint64_t last_time = 0;
uint64_t lap = stm_laptime(&last_time);  // 0 on first call
double dt = stm_sec(lap);

// --- Smooth frame time for display ---
uint64_t snapped = stm_round_to_common_refresh_rate(lap);
double fps = 1.0 / stm_sec(snapped);
```

```c
    void* user_data            // user data (unused with slog_func)
);
```

## 7. sokol-shdc Shader Workflow

### 7.1 GLSL Source (input.glsl)

```glsl
// Uniform blocks with @block
@block shared_uniforms
uniform shared_params {
    mat4 view_proj;
};
@end

// Vertex shader
@vs vs
@include_block shared_uniforms

in vec3 position;
in vec2 texcoord0;
in vec3 normal;

out vec2 uv;
out vec3 norm;

uniform vs_params {
    mat4 model;
};

void main() {
    gl_Position = view_proj * model * vec4(position, 1.0);
    uv = texcoord0;
    norm = mat3(model) * normal;
}
@end

// Fragment shader
@fs fs
in vec2 uv;
in vec3 norm;
out vec4 frag_color;

uniform texture2D tex;
uniform sampler smp;

uniform fs_params {
    vec3 light_dir;
    float ambient;
};

void main() {
    float NdotL = max(dot(normalize(norm), light_dir), 0.0);
    vec4 tex_color = texture(sampler2D(tex, smp), uv);
    frag_color = tex_color * (ambient + NdotL);
}
@end

// Program = vertex + fragment
@program my_shader vs fs
```

### 7.2 sokol-shdc CLI

```bash
sokol-shdc \
    --input  shaders/my_shader.glsl \
    --output shaders/my_shader.glsl.h \
    --slang  glsl430           # OpenGL 4.3 GLSL (or glsl330 for 3.3)
    # --slang glsl430:hlsl5:metal_macos:wgpu  (multi-backend)
```

### 7.3 Generated Header Usage

```c
#include "shaders/my_shader.glsl.h"

// Create shader using generated descriptor function:

// - Uniform block C structs:  vs_params_t, fs_params_t
// - Bind slot constants:      SG_SLOT_vs_params, SG_SLOT_tex, SG_SLOT_smp
// - Vertex attr slot constants: SG_ATTR_vs_position, SG_ATTR_vs_texcoord0, etc.
```

### 7.4 Using Generated Shader in Pipeline

```c
sg_pipeline pip = sg_make_pipeline(&(sg_pipeline_desc){
    .shader = shd,
    .layout = {
        .attrs = {
            [SG_ATTR_vs_position]  = { .format = SG_VERTEXFORMAT_FLOAT3 },
            [SG_ATTR_vs_texcoord0] = { .format = SG_VERTEXFORMAT_FLOAT2 },
            [SG_ATTR_vs_normal]    = { .format = SG_VERTEXFORMAT_FLOAT3 },
        },
    },
    .depth = {
        .compare = SG_COMPAREFUNC_LESS_EQUAL,

    },
    .cull_mode = SG_CULLMODE_BACK,
});
```


### 8.10 Full Render Loop Pattern

```c
void frame(void) {
    // 1. Compute delta time
    uint64_t dt_ticks = stm_laptime(&state.last_time);
    float dt = (float)stm_sec(dt_ticks);

    // 2. Update game state
    update_game(dt);

    // 3. Begin default pass
    sg_begin_pass(&(sg_pass){
        .action = state.pass_action,
        .swapchain = sglue_swapchain(),
    });

    // 4. For each material/shader group:
    sg_apply_pipeline(state.pip);
    sg_apply_bindings(&state.bind);

    // 5. Upload uniforms
    vs_params_t vs_p = { .mvp = compute_mvp() };
    sg_apply_uniforms(SG_SLOT_vs_params, &SG_RANGE(vs_p));

    // 6. Draw
    sg_draw(0, state.num_elements, 1);

    // 7. End pass + commit
    sg_end_pass();
    sg_commit();
}
```

### 8.11 Input Handling Pattern

```c
void event(const sapp_event* ev) {
    switch (ev->type) {
    case SAPP_EVENTTYPE_KEY_DOWN:
        if (ev->key_code == SAPP_KEYCODE_ESCAPE) sapp_request_quit();
        if (ev->key_code == SAPP_KEYCODE_F11)    sapp_toggle_fullscreen();
        if (ev->key_code == SAPP_KEYCODE_W)       input.forward = true;
        break;
    case SAPP_EVENTTYPE_KEY_UP:
        if (ev->key_code == SAPP_KEYCODE_W)       input.forward = false;
        break;
    case SAPP_EVENTTYPE_MOUSE_MOVE:
        if (sapp_mouse_locked()) {
            camera.yaw   += ev->mouse_dx * sensitivity;
            camera.pitch += ev->mouse_dy * sensitivity;
        }
        break;
    case SAPP_EVENTTYPE_MOUSE_DOWN:
        if (ev->mouse_button == SAPP_MOUSEBUTTON_LEFT) {
            sapp_lock_mouse(true);
        }
        break;
    case SAPP_EVENTTYPE_MOUSE_SCROLL:
        camera.fov -= ev->scroll_y * 2.0f;
        break;
    case SAPP_EVENTTYPE_RESIZED:
        // framebuffer automatically resized
        break;
    case SAPP_EVENTTYPE_QUIT_REQUESTED:
        // sapp_cancel_quit() to prevent, or let it proceed
        break;
    default:
        break;
    }
}
```

### 8.12 Instanced Rendering

```c
// Instance data buffer
typedef struct { float x, y, z, scale; } instance_t;
sg_buffer inst_buf = sg_make_buffer(&(sg_buffer_desc){
    .size = MAX_INSTANCES * sizeof(instance_t),
    .usage = { .vertex_buffer = true, .dynamic_update = true },

// Pipeline with two vertex buffer slots

    .shader = shd,
    .layout = {
        .buffers = {
            [0] = { .stride = sizeof(vertex_t) },
            [1] = { .stride = sizeof(instance_t), .step_func = SG_VERTEXSTEP_PER_INSTANCE },
        },
        .attrs = {
            [SG_ATTR_vs_position]  = { .buffer_index = 0, .format = SG_VERTEXFORMAT_FLOAT3 },
            [SG_ATTR_vs_texcoord0] = { .buffer_index = 0, .format = SG_VERTEXFORMAT_FLOAT2 },
            [SG_ATTR_vs_inst_pos]  = { .buffer_index = 1, .format = SG_VERTEXFORMAT_FLOAT3 },
            [SG_ATTR_vs_inst_scale]= { .buffer_index = 1, .format = SG_VERTEXFORMAT_FLOAT },
        },
    },
    .index_type = SG_INDEXTYPE_UINT16,
    .depth.compare = SG_COMPAREFUNC_LESS_EQUAL,
    .depth.write_enabled = true,
    .cull_mode = SG_CULLMODE_BACK,
});

// Bindings
state.bind.vertex_buffers[0] = mesh_vbuf;
state.bind.vertex_buffers[1] = inst_buf;
```

## Quick Reference Card

### Frame Lifecycle
```
sokol_main()          → return sapp_desc
  init_cb()           → sg_setup(), create resources
  frame_cb() (loop)   → sg_begin_pass → apply → draw → sg_end_pass → sg_commit
  event_cb()          → handle sapp_event
  cleanup_cb()        → destroy resources, sg_shutdown()
```

### Draw Call Sequence (per pass)
```
sg_begin_pass(...)
  sg_apply_pipeline(pip)
  sg_apply_bindings(&bind)
  sg_apply_uniforms(slot, &SG_RANGE(params))
  sg_draw(base, count, instances)
  // repeat apply_*/draw for more objects
sg_end_pass()
```

### Resource Dependencies
```
sg_shader  ← sg_shader_desc (from sokol-shdc generated header)
sg_pipeline ← sg_shader + vertex layout + render state
sg_buffer  ← vertex/index data
sg_image   ← pixel data or render target
sg_sampler ← filter/wrap settings
sg_view    ← image/buffer for binding to shader
sg_bindings ← buffers + views + samplers
sg_pass    ← pass_action + (swapchain OR attachments)
```
